df_pivot = df.pivot_table(
    columns='coffee_name',
    index='Month_name',
    values='money',
    aggfunc='sum',
)

df_pivot['month_number'] = pd.to_datetime(df_pivot['Month_name'], format='%b').dt.month
df_pivot.sort_values('month_number', inplace=True)
df_pivot.set_index('Month_name', inplace=True)
